### 2. Import styles:

```ts
import "superkey/styles.css";
```

### 3. How to use:

```jsx
import { Command, CommandInput, CommandList, CommandOption } from "superkey";

function App() {
  return (
    <Command open={true}>
      <CommandInput onChange={(e) => {
        console.log(e.target.value);
      }} />
      <CommandList>
        <CommandOption value="Option 1">
          <h1>Option 1</h1>
          <p>Description</p>
        </CommandOption>
        <CommandOption value="Option 2">
          <h1>Option 2</h1>
          <p>Description</p>
        </CommandOption>
      </CommandList>
    </Command>
  );
}
```
